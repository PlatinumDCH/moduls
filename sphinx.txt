"""
sys.path добавляется в обсолютный путь до родительской директории, чтобы иметь
возможность нахоидить модули на уровень выше 
"""

import sys
import os
sys.path.append(os.path.abspath('..')) 

project = 'moduls' #название проекта
copyright = '2024, dima' #авторские права
author = 'dima'
release = '0.0.1'
extensions = ['sphinx.ext.autodoc']

templates_path = ['_templates'] #путь к шаблонам документации
exclude_patterns = ['_build', 'Thumbs.db', '.DS_Store']#эти не будут включены

language = 'ru'
html_theme = 'nature' #тема оформления
html_static_path = ['_static'] #путь к статическим файлам.Напрмер измображением
#которые будет использоватся в документации

docs/index.rst
тут прописано к каким папкам проетка нужно создать документацию
пример одного пути к такому файлу



Name Project repository Notes [1]
=============================
.. automodule :: path/to/module/file  [2]
   :members:   [3]
   :undoc-members: [4]
   :show-inheritance:  [5]
(создает документацию для модуля notes.py) c подробным описанием всех
функций и классов включительно наследование

[1]оглавление написанно в стиле  ReStructuredText
====== - это росделочаная линия.Она не может быть меньшей за оглавление
[2] - дериватиа sphinx  автоматически создает документация для file.py
дополнительные директива
[3]указывает что все функции и классы должны бть вклюяены в документоцию
[4]указывает что документация так же будет для функция и методов без докстринга
[5]наследование классов должно быть включено в документацию

в конце файла
:ref:genindex - генерирует индекс что вмещает списко всех терминов в документации
:ref:modindex - генерирует индекс всех модулей документированых в докементации
:ref:search - создает форму поиска, которую пользователь может использовать для
поиска за ключевыми словами в документации.Поиск по документации.
